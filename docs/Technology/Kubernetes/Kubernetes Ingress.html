使用Ingress ​

代理一个服务 ​

bash
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: hexo-ingress-https
  namespace: senna-server
  annotations:
    kubernetes.io/ingress.class: "nginx"
spec:
  tls:
    - hosts:
      - www.senhao.top
      - senhao.top
      #使用tls
      secretName: tls-secret
  rules:
    - host: senhao.top
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: hexo
                port:
                  number: 80
    - host: www.senhao.top
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: hexo
                port:
                  number: 80

安装tls ​

bash
kubectl create secret tls tls-secret \
    --namespace senna-server \
    --key tls.key \
    --cert tls.crt
